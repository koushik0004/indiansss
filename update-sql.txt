ALTER TABLE `journals` ADD `tags` TEXT NULL DEFAULT NULL AFTER `upload_path`;
CREATE FULLTEXT INDEX fulltext_tags ON  journals(tags);
CREATE INDEX index_title ON journals(title);
CREATE INDEX index_written_by ON journals(written_by);
UPDATE `journals` jr SET jr.tags = CONCAT(jr.title, ', ', jr.written_by) WHERE 1;

CREATE TABLE `makemarr_etravers`.`submitted_manuscripts` ( `id` BIGINT(20) NOT NULL AUTO_INCREMENT , `title` VARCHAR(255) NOT NULL , `content` TEXT NOT NULL , 
`written_by` VARCHAR(255) NOT NULL , `upload_path` VARCHAR(255) NOT NULL , `tags` TEXT NOT NULL , `isblocked` ENUM('0','1') 
NOT NULL DEFAULT '0' , `isdeleted` ENUM('0','1') NOT NULL DEFAULT '0' , PRIMARY KEY (`id`)) ENGINE = InnoDB;
